Filter by Checkboxes Code

Controller


'use strict';

var controllers = angular.module('app.controllers',[]);

controllers.controller('MainCtrl', function( $scope, $location, $routeParams, Host, $filter)  {
  $scope.filterItems = {
    'Dev': true,
    'Staging': true,
    'Production': true
  };
  $scope.items = [
    {name:'Dev'}, 
    {name:'Staging'}, 
    {name:'Production'}
  ];
  //filter
  $scope.testFilter = function (host) {
    return $scope.filterItems[host.environment.title];
  };
});

HTML
